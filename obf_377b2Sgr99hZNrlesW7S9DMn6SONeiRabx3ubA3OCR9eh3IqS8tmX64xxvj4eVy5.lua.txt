local function getNearestPlayer()
	local nearestPlayer = nil;
	local shortestDistance = math.huge;
	for _, plr in ipairs(Players:GetPlayers()) do
		if ((plr ~= player) and plr.Character) then
			local plrHrp = plr.Character:FindFirstChild(LUAOBFUSACTOR_DECRYPT_STR_0("\40\102\15\126\69\1\131\4\65\13\112\95\62\139\18\103", "\234\96\19\98\31\43\110"));
			if (plrHrp and hrp) then
				local distance = (hrp.Position - plrHrp.Position).Magnitude;
				if (distance < shortestDistance) then
					shortestDistance = distance;
					nearestPlayer = plr;
				end
			end
		end
	end
	return nearestPlayer;
end

local function executeSteal(prompt, name)
	if isStealing then
		return;
	end
	if not StealData[prompt] then
		StealData[prompt] = {[LUAOBFUSACTOR_DECRYPT_STR_0("\247\126\88\178", "\156\159\17\52\214\86\190")]={},[LUAOBFUSACTOR_DECRYPT_STR_0("\186\253\180\187\169\234\175", "\220\206\143\221")]={},[LUAOBFUSACTOR_DECRYPT_STR_0("\148\120\44\19\193", "\178\230\29\77\119\184\172")]=true};
		pcall(function()
			if getconnections then
				for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
					if c.Function then
						table.insert(StealData[prompt].hold, c.Function);
					end
				end
				for _, c in ipairs(getconnections(prompt.Triggered)) do
					if c.Function then
						table.insert(StealData[prompt].trigger, c.Function);
					end
				end
			end
		end);
	end
	local data = StealData[prompt];
	if not data.ready then
		return;
	end
	data.ready = false;
	isStealing = true;
	stealStartTime = tick();
	if ProgressLabel then
		ProgressLabel.Text = name or LUAOBFUSACTOR_DECRYPT_STR_0("\198\138\47\58\91\209\219\153\68\85\57", "\152\149\222\106\123\23");
	end
	if progressConnection then
		progressConnection:Disconnect();
	end
	progressConnection = RunService.Heartbeat:Connect(function()
		if not isStealing then
			progressConnection:Disconnect();
			return;
		end
		local elapsed = tick() - stealStartTime;
		local percent = math.min((elapsed / Values.STEAL_DURATION) * 100, 100);
		if ProgressBarFill then
			ProgressBarFill.Size = UDim2.new(percent / 100, 0, 1, 0);
		end
		if ProgressPercentLabel then
			ProgressPercentLabel.Text = getDiscordProgress(percent);
		end
		if (percent >= 100) then
			progressConnection:Disconnect();
		end
	end);
	pcall(function()
		for _, fn in ipairs(data.hold) do
			fn(prompt);
		end
	end);
	task.wait(Values.STEAL_DURATION);
	pcall(function()
		for _, fn in ipairs(data.trigger) do
			fn(prompt);
		end
	end);
	task.wait(0.3);
	isStealing = false;
	data.ready = true;
	ResetProgressBar();
end
function startAutoSteal()
	task.spawn(function()
		while Enabled.AutoSteal do
			if selectedBrainrot then
				local brainrotData = detectedBrainrots[selectedBrainrot];
				if (brainrotData and brainrotData.prompt) then
					executeSteal(brainrotData.prompt, brainrotData.displayName);
				else
					local allBrainrots = findAllBrainrots();
					for _, data in ipairs(allBrainrots) do
						if (data.part == selectedBrainrot) then
							detectedBrainrots[selectedBrainrot] = data;
							executeSteal(data.prompt, data.displayName);
							break;
						end
					end
				end
			else
				local prompt, distance, name = findNearestPrompt();
				if (prompt and distance) then
					executeSteal(prompt, name);
				end
			end
			task.wait(0.5);
		end
	end);
end
function stopAutoSteal()
	Enabled.AutoSteal = false;
	isStealing = false;
	if progressConnection then
		progressConnection:Disconnect();
	end
	ResetProgressBar();
end
local function makeDraggable(frame, dragHandle)
	local dragging;
	local dragInput;
	local dragStart;
	local startPos;
	local function update(input)
		local delta = input.Position - dragStart;
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y);
	end
	dragHandle.InputBegan:Connect(function(input)
		if ((input.UserInputType == Enum.UserInputType.MouseButton1) or (input.UserInputType == Enum.UserInputType.Touch)) then
			dragging = true;
			dragStart = input.Position;
			startPos = frame.Position;
			input.Changed:Connect(function()
				if (input.UserInputState == Enum.UserInputState.End) then
					dragging = false;
				end
			end);
		end
	end);
	dragHandle.InputChanged:Connect(function(input)
		if ((input.UserInputType == Enum.UserInputType.MouseMovement) or (input.UserInputType == Enum.UserInputType.Touch)) then
			dragInput = input;
		end
	end);
	UserInputService.InputChanged:Connect(function(input)
		if ((input == dragInput) and dragging) then
			update(input);
		end
	end);
end

local function updateBestPet()

	bestPetModel = nil;
	for _, petData in ipairs(allPets) do
		detectedPets[petData.part] = {[LUAOBFUSACTOR_DECRYPT_STR_0("\174\66\8\235", "\142\192\35\101")]=petData.displayName,[LUAOBFUSACTOR_DECRYPT_STR_0("\192\116\37\182\226", "\118\182\21\73\195\135\236\204")]=petData.value,[LUAOBFUSACTOR_DECRYPT_STR_0("\24\61\8\84", "\157\104\92\122\32\100\109")]=petData.part};
		if (petData.value > bestValue) then
			bestValue = petData.value;
			bestPetModel = petData.part;
		end
	end
	updateAdvancedTpPetListGUI(allPets);
	updateAdvancedTpTargetDisplay();
end

		valueLabel.Text = formatMoney(brainrotData.value);
		valueLabel.TextColor3 = ((i == 1) and Color3.fromRGB(85, 255, 127)) or Color3.fromRGB(255, 215, 85);
		valueLabel.TextSize = 11 * scaleFactor;
		valueLabel.Font = Enum.Font.GothamBold;
		valueLabel.TextXAlignment = Enum.TextXAlignment.Right;
		entry.MouseButton1Click:Connect(function()
			if (selectedBrainrot == brainrotData.part) then
				selectedBrainrot = nil;
			else
				selectedBrainrot = brainrotData.part;
			end
			updateBrainrotTargetDisplay();
			updateBrainrotListGUI(brainrotList);
		end);
		entry.MouseEnter:Connect(function()
			if not isTarget then
				entry.BackgroundColor3 = Color3.fromRGB(45, 45, 50);
			end
		end);
		entry.MouseLeave:Connect(function()
			if not isTarget then
				entry.BackgroundColor3 = (((i % 2) == 0) and Color3.fromRGB(35, 35, 40)) or Color3.fromRGB(30, 30, 35);
			end
		end);
	end
	local contentSize = brainrotListLayout.AbsoluteContentSize;
	brainrotScrollFrame.CanvasSize = UDim2.new(0, 0, 0, contentSize.Y + 10);
end

	detectedBrainrots = {};

		detectedBrainrots[brainrotData.part] = brainrotData;
	end
	updateBrainrotListGUI(allBrainrots);
	updateBrainrotTargetDisplay();
end
