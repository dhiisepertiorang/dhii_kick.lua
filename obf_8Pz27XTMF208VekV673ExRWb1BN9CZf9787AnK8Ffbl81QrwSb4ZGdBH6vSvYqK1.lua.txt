local function getNearestPlayer()
	local nearestPlayer = nil;
	local shortestDistance = math.huge;
	for _, plr in ipairs(Players:GetPlayers()) do
		if ((plr ~= player) and plr.Character) then
			local plrHrp = plr.Character:FindFirstChild(LUAOBFUSACTOR_DECRYPT_STR_0("\168\233\143\119\62\231\191\18\178\243\141\98\0\233\164\2", "\118\224\156\226\22\80\136\214"));
			if (plrHrp and hrp) then
				local distance = (hrp.Position - plrHrp.Position).Magnitude;
				if (distance < shortestDistance) then
					shortestDistance = distance;
					nearestPlayer = plr;
				end
			end
		end
	end
	return nearestPlayer;
end

local function executeSteal(prompt, name)
	if isStealing then
		return;
	end
	if not StealData[prompt] then
		StealData[prompt] = {[LUAOBFUSACTOR_DECRYPT_STR_0("\208\73\12\119", "\203\184\38\96\19\203")]={},[LUAOBFUSACTOR_DECRYPT_STR_0("\45\97\112\70\201\60\97", "\174\89\19\25\33")]={},[LUAOBFUSACTOR_DECRYPT_STR_0("\61\23\83\74\238", "\107\79\114\50\46\151\231")]=true};
		pcall(function()
			if getconnections then
				for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
					if c.Function then
						table.insert(StealData[prompt].hold, c.Function);
					end
				end
				for _, c in ipairs(getconnections(prompt.Triggered)) do
					if c.Function then
						table.insert(StealData[prompt].trigger, c.Function);
					end
				end
			end
		end);
	end
	local data = StealData[prompt];
	if not data.ready then
		return;
	end
	data.ready = false;
	isStealing = true;
	stealStartTime = tick();
	if ProgressLabel then
		ProgressLabel.Text = name or LUAOBFUSACTOR_DECRYPT_STR_0("\10\146\144\8\166\16\153\231\119\232\251", "\160\89\198\213\73\234\89\215");
	end
	if progressConnection then
		progressConnection:Disconnect();
	end
	progressConnection = RunService.Heartbeat:Connect(function()
		if not isStealing then
			progressConnection:Disconnect();
			return;
		end
		local elapsed = tick() - stealStartTime;
		local percent = math.min((elapsed / Values.STEAL_DURATION) * 100, 100);
		if ProgressBarFill then
			ProgressBarFill.Size = UDim2.new(percent / 100, 0, 1, 0);
		end
		if ProgressPercentLabel then
			ProgressPercentLabel.Text = getDiscordProgress(percent);
		end
		if (percent >= 100) then
			progressConnection:Disconnect();
		end
	end);
	pcall(function()
		for _, fn in ipairs(data.hold) do
			fn(prompt);
		end
	end);
	task.wait(Values.STEAL_DURATION);
	pcall(function()
		for _, fn in ipairs(data.trigger) do
			fn(prompt);
		end
	end);
	task.wait(0.3);
	isStealing = false;
	data.ready = true;
	ResetProgressBar();
end
function startAutoSteal()
	task.spawn(function()
		while Enabled.AutoSteal do
			if selectedBrainrot then
				local brainrotData = detectedBrainrots[selectedBrainrot];
				if (brainrotData and brainrotData.prompt) then
					executeSteal(brainrotData.prompt, brainrotData.displayName);
				else
					local allBrainrots = findAllBrainrots();
					for _, data in ipairs(allBrainrots) do
						if (data.part == selectedBrainrot) then
							detectedBrainrots[selectedBrainrot] = data;
							executeSteal(data.prompt, data.displayName);
							break;
						end
					end
				end
			else
				local prompt, distance, name = findNearestPrompt();
				if (prompt and distance) then
					executeSteal(prompt, name);
				end
			end
			task.wait(0.5);
		end
	end);
end
function stopAutoSteal()
	Enabled.AutoSteal = false;
	isStealing = false;
	if progressConnection then
		progressConnection:Disconnect();
	end
	ResetProgressBar();
end
local function makeDraggable(frame, dragHandle)
	local dragging;
	local dragInput;
	local dragStart;
	local startPos;
	local function update(input)
		local delta = input.Position - dragStart;
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y);
	end
	dragHandle.InputBegan:Connect(function(input)
		if ((input.UserInputType == Enum.UserInputType.MouseButton1) or (input.UserInputType == Enum.UserInputType.Touch)) then
			dragging = true;
			dragStart = input.Position;
			startPos = frame.Position;
			input.Changed:Connect(function()
				if (input.UserInputState == Enum.UserInputState.End) then
					dragging = false;
				end
			end);
		end
	end);
	dragHandle.InputChanged:Connect(function(input)
		if ((input.UserInputType == Enum.UserInputType.MouseMovement) or (input.UserInputType == Enum.UserInputType.Touch)) then
			dragInput = input;
		end
	end);
	UserInputService.InputChanged:Connect(function(input)
		if ((input == dragInput) and dragging) then
			update(input);
		end
	end);
end

		if (petData.value > bestValue) then
			bestValue = petData.value;
			bestPetModel = petData.part;
		end
	end
	updateAdvancedTpPetListGUI(allPets);
	updateAdvancedTpTargetDisplay();
end
